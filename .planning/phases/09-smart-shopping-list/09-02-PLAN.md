**Check-off handler (`handleCheckOff`):**
```typescript
async function handleCheckOff(item: ShoppingListItem) {
  // 1. Optimistic remove
  removeShoppingItem(item.id);
  try {
    // 2. Delete from shopping_list_items
    await supabase.from('shopping_list_items').delete().eq('id', item.id);
    // 3. Restore pantry quantity if pantry_item_id is set
    if (item.pantry_item_id) {
      // fetch low_stock_threshold for this item
      const { data: pantryRow } = await supabase
        .from('pantry_items')
        .select('low_stock_threshold')
        .eq('id', item.pantry_item_id)
        .single();
      const restoreQty = pantryRow?.low_stock_threshold ?? 1;
      await supabase
        .from('pantry_items')
        .update({ quantity: restoreQty })
        .eq('id', item.pantry_item_id);
      // Also update the pantry store in memory (per D-03)
      updateItem(item.pantry_item_id, { quantity: restoreQty });
    }
  } catch {
    // Re-add the item on error
    addShoppingItem(item);
    showAlert(t('pantry.error_save_title'), t('pantry.shop_error_checkoff'));
  }
}
```

```typescript
async function handleExport() {
  const sorted = [...shoppingItems].sort((a, b) => a.name.localeCompare(b.name));
  const lines = sorted.map((item) => {
    if (item.quantity != null && item.unit) {
      return `- ${item.name} × ${item.quantity} ${item.unit}`;
    }
    return `- ${item.name}`;
  });
  const message = `Liste de courses Ziko\n\n${lines.join('\n')}`;
  await Share.share({ message });
}
```

```typescript
async function handleAddToList() {
  try {
    const { data: { user } } = await supabase.auth.getUser();
    if (!user) return;

    // Fetch current shopping list (in case store is stale)
    const { data: existingList } = await supabase
      .from('shopping_list_items')
      .select('name')
      .eq('user_id', user.id);

    const existingNames = new Set(
      (existingList ?? []).map((r: { name: string }) => r.name.toLowerCase())
    );

    // Filter recipe ingredients: only those not in pantry (by name, case-insensitive)
    const pantryNames = new Set(pantryItems.map((p) => p.name.toLowerCase()));
    const missing = recipe.ingredients.filter(
      (ing) => !pantryNames.has(ing.name.toLowerCase())
    );

    // Deduplicate against existing shopping list (D-05)
    const toInsert = missing.filter(
      (ing) => !existingNames.has(ing.name.toLowerCase())
    );

    if (toInsert.length === 0) return; // all already in list or all in pantry — silent

    const rows = toInsert.map((ing) => ({
      user_id: user.id,
      name: ing.name,
      quantity: ing.quantity ?? null,
      unit: ing.unit ?? null,
      pantry_item_id: null,
      source: 'recipe' as const,
      recipe_name: recipe.name,
    }));

    const { data: inserted, error } = await supabase
      .from('shopping_list_items')
      .insert(rows)
      .select();

    if (error) throw error;

    // Update store
    if (inserted) {
      inserted.forEach((item: any) => addShoppingItem(item));
    }
  } catch {
    showAlert(t('pantry.error_save_title'), t('pantry.shop_error_add'));
  }
}
```
